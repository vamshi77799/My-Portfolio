import pandas as pd
import matplotlib.pyplot as plt
import math as mp

# --- Load Data ---
df = pd.read_csv("NACA 4412 copy.csv", header=5)

def rotate(x_list, y_list, angle_deg):
    rad = mp.radians(angle_deg)
    new_x, new_y = [], []
    for i, m in zip(x_list, y_list):
        v = mp.cos(rad)*i - mp.sin(rad)*m
        w = mp.sin(rad)*i + mp.cos(rad)*m
        new_x.append(v)
        new_y.append(w)
    return new_x, new_y

# Extract Columns
x, y = df['X'], df["Yc"]
cx, cy = df["Chord X"], df["Chord Y"]
xu, yu = df["xu"], df["yu"]
xl, yl = df["xl"], df["yl"]

# --- Calculate Rotations (AOA) ---
x1, y1 = rotate(x, y, -5)
xu1, yu1 = rotate(xu, yu, -5)
xl1, yl1 = rotate(xl, yl, -5)
cx1, cy1 = rotate(cx, cy, -5)

x2, y2 = rotate(x, y, -10)
xu2, yu2 = rotate(xu, yu, -10)
xl2, yl2 = rotate(xl, yl, -10)
cx2, cy2 = rotate(cx, cy, -10)

x3, y3 = rotate(x, y, -15)
xu3, yu3 = rotate(xu, yu, -15)
xl3, yl3 = rotate(xl, yl, -15)
cx3, cy3 = rotate(cx, cy, -15)

# --- Plotting ---
plt.close('all')
fig, a = plt.subplots(nrows=2, ncols=2, figsize=(24, 14), dpi=100)
plt.subplots_adjust(hspace=0.3, wspace=0.2)

plots = [
    (a[0,0], "NACA 4412 (AOA 0°)", x, y, cx, cy, xu, yu, xl, yl),
    (a[0,1], "NACA 4412 (AOA 5°)", x1, y1, cx1, cy1, xu1, yu1, xl1, yl1),
    (a[1,0], "NACA 4412 (AOA 10°)", x2, y2, cx2, cy2, xu2, yu2, xl2, yl2),
    (a[1,1], "NACA 4412 (AOA 15°)", x3, y3, cx3, cy3, xu3, yu3, xl3, yl3)
]

for ax, title, vx, vy, vcx, vcy, vxu, vyu, vxl, vyl in plots:
    ax.set_aspect('equal', adjustable='datalim') 
    ax.set_title(title, fontsize=18, color='red')
    ax.plot(vx, vy, '--', markersize=3, c='black', label="Mean Camber")
    ax.plot(vcx, vcy, 'r--', label="Chord Line")
    ax.plot(vxu, vyu, 'g', label="Upper Surface")
    ax.plot(vxl, vyl, 'b', label="Lower Surface")
    ax.legend(fontsize=10, loc='upper right')
    ax.grid(True, alpha=0.3)

plt.show()

import pyaudio
import numpy as np
import matplotlib.pyplot as plt
import matplotlib.animation as animation

# --- Configuration ---
CHUNK = 1024       # Audio chunk size
RATE = 44100       # Sampling rate
HISTORY_SIZE = 300 # How many frames to keep in history

# --- Setup PyAudio ---
p = pyaudio.PyAudio()
stream = p.open(format=pyaudio.paInt16,
                channels=1,
                rate=RATE,
                input=True,
                frames_per_buffer=CHUNK)

# --- Setup Plotting (Dark Theme) ---
plt.style.use('dark_background')
fig, ax = plt.subplots(figsize=(12, 6))

# Residual History Lists
bass_hist, mid_hist, high_hist = [], [], []

# Lines
line_bass, = ax.plot([], [], color='#00FFFF', lw=1.0, label='Bass')
line_mid,  = ax.plot([], [], color='#FF00FF', lw=1.0, label='Mids')
line_high, = ax.plot([], [], color='#FFFF00', lw=1.0, label='Highs')

# Styling
ax.set_title('Frequency Residuals Monitor', color='white', fontsize=14)
ax.set_ylim(0, 100)
ax.set_xlim(0, HISTORY_SIZE)
ax.grid(True, color='#333', linestyle=':')
ax.legend(loc='upper right')

def update(frame):
    # Read raw audio
    data = np.frombuffer(stream.read(CHUNK), dtype=np.int16)
    
    # FFT calculation
    fft_data = np.abs(np.fft.rfft(data))
    
    # Calculate Frequency Bands
    bass = np.mean(fft_data[0:10]) / 50    # Normalize roughly
    mid  = np.mean(fft_data[10:50]) / 50
    high = np.mean(fft_data[50:200]) / 50
    
    # Update History
    bass_hist.append(bass)
    mid_hist.append(mid)
    high_hist.append(high)
    
    # Scroll Graph
    if len(bass_hist) > HISTORY_SIZE:
        bass_hist.pop(0)
        mid_hist.pop(0)
        high_hist.pop(0)
        
    # Set Data
    line_bass.set_data(range(len(bass_hist)), bass_hist)
    line_mid.set_data(range(len(mid_hist)), mid_hist)
    line_high.set_data(range(len(high_hist)), high_hist)
    
    return line_bass, line_mid, line_high

ani = animation.FuncAnimation(fig, update, interval=30, blit=True)
plt.show()
